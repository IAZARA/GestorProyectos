server {
    listen 80;
    server_name 137.184.198.221 dngbds.online www.dngbds.online;

    # Ruta específica para clear-storage.html
    location = /clear-storage.html {
        root /var/www/GestorProyectos/public;
        try_files /clear-storage.html =404;
    }

    # Servir archivos estáticos directamente
    location /public/ {
        alias /var/www/GestorProyectos/public/;
        try_files $uri $uri/ =404;
    }

    location /socket.io/ {
        proxy_pass http://localhost:3001;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
    }

    location / {
        proxy_pass http://localhost:3000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
    }
}
